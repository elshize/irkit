94  void assemble(std::istream& input)
95  {
96  if (!fs::exists(output_dir_)) { fs::create_directory(output_dir_); }
97  fs::path work_dir = output_dir_ / ".batches";
98  if (!fs::exists(work_dir)) { fs::create_directory(work_dir); }
99 
100  int batch_number = 0;
101  std::vector<fs::path> batch_dirs;
102  while (input && input.peek() != EOF)
103  {
104  std::clog << "Building batch " << batch_number << std::endl;
105  fs::path batch_dir = work_dir / std::to_string(batch_number);
106  metadata batch_metadata(batch_dir);
107  build_batch(input, batch_metadata);
108  batch_dirs.push_back(std::move(batch_dir));
109  ++batch_number;
110  }
111  std::clog << "Merging " << batch_number << " batches" << std::endl;
112  irk::index_merger<long, std::string, long, long> merger(output_dir_,
113  batch_dirs,
114  document_codec_,
115  frequency_codec_,
116  block_size_);
117  merger.merge();
118  auto term_map = irk::build_prefix_map_from_file<long>(
119  irk::index::terms_path(output_dir_));
120  irk::io::dump(term_map, irk::index::term_map_path(output_dir_));
121  auto title_map = irk::build_prefix_map_from_file<long>(
122  irk::index::titles_path(output_dir_));
123  irk::io::dump(title_map, irk::index::title_map_path(output_dir_));
124  std::clog << "Success!" << std::endl;
125  }

132  std::istream&
133  build_batch(std::istream& input, metadata batch_metadata) const
134  {
135  if (!fs::exists(batch_metadata.dir))
136  {
137  fs::create_directory(batch_metadata.dir);
138  }
139 
140  std::ofstream of_doc_ids(batch_metadata.doc_ids.c_str());
141  std::ofstream of_doc_ids_off(batch_metadata.doc_ids_off.c_str());
142  std::ofstream of_doc_counts(batch_metadata.doc_counts.c_str());
143  std::ofstream of_doc_counts_off(batch_metadata.doc_counts_off.c_str());
144  std::ofstream of_terms(batch_metadata.terms.c_str());
145  std::ofstream of_term_doc_freq(batch_metadata.term_doc_freq.c_str());
146  std::ofstream of_titles(batch_metadata.doc_titles.c_str());
147  std::ofstream of_sizes(batch_metadata.doc_sizes.c_str());
148  std::ofstream of_term_occurrences(
149  batch_metadata.term_occurrences.c_str());
150  std::ofstream of_properties(batch_metadata.properties.c_str());
151 
152  builder_type builder(block_size_);
153  std::string line;
154  for (int processed_documents_ = 0;
155  processed_documents_ < batch_size_;
156  processed_documents_++)
157  {
158  if (!std::getline(input, line)) { break; }
159  document_type doc = document_type(processed_documents_);
160  builder.add_document(doc++);
161  std::istringstream linestream(line);
162  std::string title;
163  linestream >> title;
164  of_titles << title << '\n';
165  std::string term;
166  while (linestream >> term) { builder.add_term(term); }
167  }
168 
169  builder.sort_terms();
170  builder.write_terms(of_terms);
171  builder.write_document_frequencies(of_term_doc_freq);
172  builder.write_document_ids(of_doc_ids, of_doc_ids_off);
173  builder.write_document_counts(of_doc_counts, of_doc_counts_off);
174  builder.write_document_sizes(of_sizes);
175  builder.write_term_occurrences(of_term_occurrences);
176  builder.write_properties(of_properties);
177  of_titles.close();
178  of_terms.close();
179  auto term_map = irk::build_prefix_map_from_file<long>(
180  irk::index::terms_path(batch_metadata.dir));
181  irk::io::dump(term_map, irk::index::term_map_path(batch_metadata.dir));
182  auto title_map = irk::build_prefix_map_from_file<long>(
183  irk::index::titles_path(batch_metadata.dir));
184  irk::io::dump(
185  title_map, irk::index::title_map_path(batch_metadata.dir));
186 
187  return input;
188  }
